limit 后面2个数字 第一个表示从第几条记录开始（基数为0）

心得体会：遇到不懂或者难解的题，不要急于求成，先要笨的方法：先把问题一步一步的分解出来，一个个条件的满足，可以先考虑一个个表的去查询结果，再把查询的结果嵌套，组织起来就能得到想要的结果了。通过这种方式，看下能不能不用嵌套的方法，再考虑采用多表查询的方式查找结果，这样或者就容易想到解决问题的方法。

select * from score,course,stu where score.stu_id = stu.stu_id and course.kc_id = score.kc_id;

select * from course where not exists (select * from score,course,stu where score.stu_id = stu.stu_id and course.kc_id = score.kc_id);

select name from stu where not exists (select * from course where not exists (select * from score,course,stu where score.stu_id = stu.stu_id and course.kc_id = score.kc_id));



select kc_id from course where not kc_id in (select course.kc_id from score,course,stu where score.stu_id = stu.stu_id and course.kc_id = score.kc_id);

select kc_id from course where kc_id in (select course.kc_id from score,course,stu where score.stu_id = stu.stu_id and course.kc_id = score.kc_id);
